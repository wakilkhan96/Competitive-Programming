#include<bits/stdc++.h>

class Solution
{
public:
    void printFloydTriangle(int N)
    {
        // Write Your Code here
        int v = 1;
        for(int i=0; i<N; i++) {
            for(int j=0; j<=i; j++) cout<<v++<<" ";
            cout<<endl;
        }
    }
};

// { Driver Code Starts.
int main()
{
    int t;
    cin >> t;
    while (t--)
    {
        int N;
        cin >> N;
        Solution ob;
        ob.printFloydTriangle(N);
    }
    return 0;
}  // } Driver Code Ends
