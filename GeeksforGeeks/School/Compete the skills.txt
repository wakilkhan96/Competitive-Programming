class Solution{
    public:
    void scores(long long a[], long long b[], int &ca, int &cb)
    {
        // Your code goes here   
        int score1(0), score2(0);
        for(int i=0; i<3; i++)
        {
            if(a[i]>b[i]) score1++;
            else if(b[i]>a[i]) score2++;
        }
        
        ca = score1, cb = score2;
    }
};

// { Driver Code Starts.
int main()
{
    int t=0;
    cin>>t;
    while(t--)
    {
        long long int a[5], b[5], i=0;
        int ca=0, cb=0;
        for(i=0; i<3; i++)
            cin>>a[i];
    
        for(i=0; i<3; i++)
            cin>>b[i];
        Solution ob;
        ob.scores(a, b, ca, cb);
        
        cout<<ca<<" "<<cb<<endl;
    }
    return 0 ;
}  // } Driver Code Ends
