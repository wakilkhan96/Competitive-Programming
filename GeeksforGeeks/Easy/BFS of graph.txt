class Solution {
  public:
    // Function to return Breadth First Traversal of given graph.
    vector<int> bfsOfGraph(int V, vector<int> adj[]) {
        // Code here
        vector<int> res, visited(V, 0);
        queue<int> q;
        q.push(0);
        
        while(!q.empty()) {
            int u = q.front(), v;
            res.push_back(u);
            q.pop();
            
            for(auto v: adj[u]) {
                if(visited[v]==0) {
                    visited[v] = 1;
                    q.push(v);
                }
            }
        }
        
        return res;
    }
};
