//Function to find the data of nth node from the end of a linked list.
int getNthFromLast(Node *head, int n)
{
       // Your code here
       if(head == NULL) return -1;
       int cou = 0;
       Node* cur = head;
       while(cur != NULL) cur = cur->next, cou++;
       n = cou-n;
       if(n<0) return -1;
       cur = head;
       while(n--) cur = cur->next;
       return cur->data;
       
}
