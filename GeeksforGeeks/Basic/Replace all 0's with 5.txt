int convertFive(int n) {
    //Your code here
    string str = to_string(n);
    n = 0;
    for(int i=0; i<str.size(); i++) {
        if(str[i]=='0') str[i] = '5'; 
        n = n*10 + (str[i]-'0');
    }
    return n;
}
