//Function to locate the occurrence of the string x in the string s.
int strstr(string s, string x)
{
    //Your code here
    for(int i=0, j=0; i<s.size(); i++) {
        j = 0;
        while(s[i+j]==x[j]) {
            j++;
            if(j==x.size()) return i;
        }
    }
    return -1;
}
