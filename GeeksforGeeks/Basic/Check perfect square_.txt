class Solution{   
public:
    int checkPerfectSquare(int N){
        // code here 
        int sqt = sqrt(N);
        return (sqt*sqt)==N;
    }
};

{

    cin>>t;
    while(t--)
    {
        int N;
        cin >> N;
        Solution ob;
        cout << ob.checkPerfectSquare(N) << endl;
    }
