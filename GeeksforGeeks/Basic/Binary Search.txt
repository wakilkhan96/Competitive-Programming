class Solution{
public:
    int binarysearch(int arr[], int n, int k){
        // code here
        int l = 0, r = n-1, m, it = 64;
        while(it--) {
            m = (l+r)/2;
            if(arr[m]==k) return m;
            else if(arr[m]<k) l = m+1;
            else r = m;
        }
        
        //if(arr[l]==k) return l;
        if(arr[m]==k) return m;
        //if(arr[r]==k) return r;
        return -1;
    }
};
